{ id: "l_ep", name: "Elevação de Quadril (Hip Thrust)", ... }

// exerciseAnimations.js
'l_ep': {
  ...gif('Weighted_Glute_Bridge'),
  muscles: ['Glúteos', 'Isquiotibiais', 'Core'],
  tip: 'Empurre com calcanhar, contraia o glúteo no topo por 2s.',
  instructions: [...]
}
